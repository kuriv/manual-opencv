多边形绘制

执行下面的代码，在原图像上绘制多边形。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img = imread("demo.jpg");
    Point p1 = Point(100, 100);
    Point p2 = Point(200, 50);
    Point p3 = Point(300, 200);
    Point p4 = Point(250, 300);
    Point p5 = Point(200, 100);
    Point p6 = Point(150, 80);
    vector<Point> p;
    p.push_back(p1);
    p.push_back(p2);
    p.push_back(p3);
    p.push_back(p4);
    p.push_back(p5);
    p.push_back(p6);
    if (img.empty())
    {
        cout << "could not load image.\n" << endl;
        return -1;
    }
    polylines(img, p, true, Scalar(255, 255, 0), 5, LINE_AA);
    fillPoly(img, p, Scalar(255, 0, 255), LINE_AA);
    imshow("IMG", img);
    waitKey(0);
    destroyAllWindows();
    return 0;
}

也可以使用下面的方式绘制并填充。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img = imread("demo.jpg");
    Point p1 = Point(100, 100);
    Point p2 = Point(200, 50);
    Point p3 = Point(300, 200);
    Point p4 = Point(250, 300);
    Point p5 = Point(200, 100);
    Point p6 = Point(150, 80);
    vector<Point> p;
    p.push_back(p1);
    p.push_back(p2);
    p.push_back(p3);
    p.push_back(p4);
    p.push_back(p5);
    p.push_back(p6);
    vector<vector<Point>> contours;
    contours.push_back(p);
    if (img.empty())
    {
        cout << "could not load image.\n" << endl;
        return -1;
    }
    drawContours(img, contours, -1, Scalar(255, 255, 0), -1, LINE_AA);
    imshow("IMG", img);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
