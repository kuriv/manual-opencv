也可以通过指针方式读取矩阵中元素的信息。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img = Mat(3, 3, CV_8UC3, Scalar(0, 0, 255));
    int rows = img.rows;
    int cols = img.cols;
    uchar *p;
    int i, j;
    for (i = 0; i < rows; i++)
    {
        p = img.ptr<uchar>(i);
        for (j = 0; j < cols; j++)
        {
            cout << (int)*p++ << (int)*p++ << (int)*p++ << endl;
        }
    }
    return 0;
}
