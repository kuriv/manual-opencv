# 数据保存

执行下面的代码，将三通道矩阵信息保存为图像。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img(512, 512, CV_8UC3, Scalar(255, 0, 255));
    imshow("IMG", img);
    if (!imwrite("demo.png", img))
    {
        cout << "Could not write image.\n" << endl;
        return -1;
    }
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```
