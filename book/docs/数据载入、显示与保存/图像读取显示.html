图像读取显示

执行下面的代码，读取指定的图像。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img = imread("demo.jpg");
    if (img.empty())
    {
        cout << "Could not load image.\n" << endl;
        return -1;
    }
    return 0;
}

执行下面的代码，为生成的窗口指定窗口名和窗口属性。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    namedWindow("IMG", WINDOW_FREERATIO);
    return 0;
}

执行下面的代码，读取并显示指定的图像。

#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img = imread("demo.jpg");
    if (img.empty())
    {
        cout << "Could not load image.\n" << endl;
        return -1;
    }
    namedWindow("IMG", WINDOW_FREERATIO);
    imshow("IMG", img);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
