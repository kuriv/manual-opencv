# 图像创建与赋值

执行下面的代码，可以克隆指定的图像矩阵。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img1, img2, img3;
    img1 = imread("demo.jpg");
    if (img1.empty())
    {
        cout << "could not load image.\n" << endl;
        return -1;
    }
    img2 = img1.clone();
    img2.copyTo(img3);
    imshow("IMG1", img1);
    imshow("IMG2", img2);
    imshow("IMG3", img3);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```

执行下面的代码，分别以单通道与三通道方式创建指定大小的图像矩阵，所有通道均赋值为 0 。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img1 = Mat::zeros(Size(5, 5), CV_8UC1);
    cout << img1 << endl;
    cout << "rows: " << img1.rows << ", cols: " << img1.cols << ", channels: " << img1.channels() << endl;
    imshow("IMG1", img1);
    Mat img2 = Mat::zeros(Size(5, 5), CV_8UC3);
    cout << img2 << endl;
    cout << "rows: " << img2.rows << ", cols: " << img2.cols << ", channels: " << img2.channels() << endl;
    imshow("IMG2", img2);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```

执行下面的代码，分别以单通道与三通道方式创建指定大小的图像矩阵，三通道中仅第一个通道被赋值为 1 ，其余通道均赋值为 0 。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img1 = Mat::ones(Size(5, 5), CV_8UC1);
    cout << img1 << endl;
    imshow("IMG1", img1);
    Mat img2 = Mat::ones(Size(5, 5), CV_8UC3);
    cout << img2 << endl;
    imshow("IMG2", img2);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```

执行下面的代码，为创建好的单通道与三通道图像矩阵手动赋值。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img1 = Mat::zeros(Size(5, 5), CV_8UC1);
    img1 = 233;
    cout << img1 << endl;
    imshow("IMG1", img1);
    Mat img2 = Mat::ones(Size(5, 5), CV_8UC3);
    img2 = Scalar(233, 233, 233);
    cout << img2 << endl;
    imshow("IMG2", img2);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```

注意，直接使用赋值操作传递的是图像矩阵的地址引用。

```cpp
#include <opencv2/opencv.hpp>
#include <iostream>

using namespace cv;
using namespace std;

int main(int argc, char const *argv[])
{
    Mat img1, img2;
    img1 = Mat::zeros(Size(5, 5), CV_8UC3);
    img1 = Scalar(233, 233, 233);
    img2 = img1;
    img2 = Scalar(0, 0, 0);  
    cout << img1 << endl;
    imshow("IMG1", img1);
    cout << img2 << endl;
    imshow("IMG2", img2);
    waitKey(0);
    destroyAllWindows();
    return 0;
}
```
